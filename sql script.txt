CREATE DATABASE  `ppl` ;

#USER
CREATE TABLE `ppl`.`user` (
`uid` MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
`user` VARCHAR(20) NOT NULL,
`pass` VARCHAR(40) NOT NULL,
`salt` VARCHAR(10) NOT NULL,
`name` VARCHAR(30),
`bday` DATE,
`about` TEXT,
`share` VARCHAR(255),
`rank` TINYINT UNSIGNED
) ENGINE = MYISAM DEFAULT CHARSET=utf8;


CREATE DATABASE  `data` ;

#MEDIA HANDLER
CREATE TABLE  `data`.`handler` (
`id` INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
`uid` MEDIUMINT UNSIGNED NOT NULL,
`title` VARCHAR (100) NOT NULL,
`desc` VARCHAR (255),
`type` VARCHAR( 5 ) NOT NULL,
`category` VARCHAR( 50 ) NOT NULL,
`tags` VARCHAR( 50 ),
`points` SMALLINT UNSIGNED NOT NULL DEFAULT 0,
`perm` TINYINT UNSIGNED NOT NULL DEFAULT 0,
`date` TIMESTAMP NOT NULL,
`img` VARCHAR (100)
#`log` VARCHAR (50)
) ENGINE = MYISAM DEFAULT CHARSET=utf8;


#VIDEO
CREATE TABLE  `data`.`video` (
`vid` INT UNSIGNED NOT NULL ,
`part` TINYINT UNSIGNED DEFAULT 1,
`title` VARCHAR( 100 ) ,
`desc` VARCHAR( 999 ) ,
`type` VARCHAR( 2 ) NOT NULL ,
`code` VARCHAR( 11 ) NOT NULL ,
`date` TIMESTAMP NOT NULL ,
`length` SMALLINT UNSIGNED,
PRIMARY KEY (  `vid` )
) ENGINE = MYISAM DEFAULT CHARSET=utf8;


#TEXTS
CREATE TABLE  `data`.`text` (
`tid` INT UNSIGNED NOT NULL ,
`part` TINYINT UNSIGNED DEFAULT 1,
`title` VARCHAR( 100 ) ,
`text` TEXT NOT NULL ,
`date` TIMESTAMP NOT NULL ,
PRIMARY KEY (  `tid` )
) ENGINE = MYISAM DEFAULT CHARSET=utf8;


#CODES
CREATE TABLE `data`.`code` (
`id` INT UNSIGNED NOT NULL ,
`file` VARCHAR(100) NOT NULL ,
PRIMARY KEY (  `id` )
) ENGINE = MYISAM DEFAULT CHARSET=utf8;


#COMMENTS
CREATE TABLE  `data`.`comments` (
`pid` INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`id` INT UNSIGNED NOT NULL ,
`comment` VARCHAR( 255 ) NOT NULL ,
`time` TIMESTAMP NOT NULL ,
`uid` MEDIUMINT UNSIGNED NOT NULL ,
INDEX (  `id` )
) ENGINE = MYISAM DEFAULT CHARSET=utf8;


CREATE DATABASE  `danmaku` ;

#DANMAKKU
CREATE TABLE  `danmaku`.`cmt#` (
`id` SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`stime` FLOAT UNSIGNED NOT NULL ,
`mode` TINYINT UNSIGNED NOT NULL ,
`size` TINYINT UNSIGNED NOT NULL ,
`color` MEDIUMINT UNSIGNED NOT NULL ,
`date` TIMESTAMP NOT NULL ,
`message` VARCHAR( 999 ) NOT NULL ,
`uid` MEDIUMINT UNSIGNED
) ENGINE = MYISAM DEFAULT CHARSET=utf8;
